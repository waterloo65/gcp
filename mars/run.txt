export PROJECT_ID=$(gcloud config get-value project)
export REGION='us-central1'
export BUCKET=gs://${PROJECT_ID}-bucket
export PIPELINE_FOLDER=${BUCKET}
export MAIN_CLASS_NAME=com.moonbank.pipeline.MarsActivitiesPipeline
export RUNNER=DataflowRunner
export INPUT_PATH=${PIPELINE_FOLDER}/mars_sample.csv
export RAW_TABLE=${PROJECT_ID}:mars.raw
export OUTPUT_TABLE=${PROJECT_ID}:mars.activities

mvn compile exec:java \
-Dexec.mainClass=${MAIN_CLASS_NAME} \
-Dexec.cleanupDaemonThreads=false \
-Dexec.args=" \
--project=${PROJECT_ID} \
--region=${REGION} \
--stagingLocation=${PIPELINE_FOLDER}/staging \
--tempLocation=${PIPELINE_FOLDER}/temp \
--runner=${RUNNER} \
--inputPath=${INPUT_PATH} \
--rawTable=${RAW_TABLE} \
--outputTable=${OUTPUT_TABLE}"
